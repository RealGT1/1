        #include<stdio.h>
        #include<stdlib.h>
            
            int main() {
                int month[12] = {31, 28, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31};
                int d, m, y, nd, nm, ny, ndays;
                char choice;
            
                while (1) {
                    while (1) {
                        int valid = 1;  
                        printf("Enter the date, month, year: ");
                        scanf("%d%d%d", &d, &m, &y);
            
                       
                        if (y < 1812 || y > 2012) {
                            printf("Invalid Input Year\n");
                            valid = 0;
                        }
            
                       
                        if (m < 1 || m > 12) {
                            printf("Invalid Input Month\n");
                            valid = 0;
                        }
            
                        
                        if (m >= 1 && m <= 12) {
                            ndays = month[m - 1];
            
                           
                            if (m == 2) {
                                if ((y % 100 == 0 && y % 400 == 0) || (y % 100 != 0 && y % 4 == 0)) {
                                    ndays = 29;
                                }
                            }
            
                            if (d <= 0 || d > ndays) {
                                printf("Invalid Input Day\n");
                                valid = 0;
                            }
                        } else if (d <= 0 || d > 31) {  
                            printf("Invalid Input Day\n");
                            valid = 0;
                        }
            
                        if (valid) {
                            break;  
                        }
                    }
            
                    nd = d + 1;
                    nm = m;
                    ny = y;
            
                    if (nd > ndays) {
                        nd = 1;
                        nm++;
                    }
            
                    if (nm > 12) {
                        nm = 1;
                        ny++;
                    }
            
                    printf("\nGiven date is %d:%d:%d", d, m, y);
                    printf("\nNext day date is %d:%d:%d", nd, nm, ny);
            
                    printf("\nDo you want to continue? (y/n): ");
                    scanf(" %c", &choice);  
            
                    if (choice != 'y' && choice != 'Y') {
                        break;
                    }
                }
            
                return 0;
            }

            int main() {
                int a, b, c;
                char choice;
            
                while (1) {
                    printf("Enter three sides of the triangle: \n");
                    scanf("%d%d%d", &a, &b, &c);
            
                    if ((a < b + c) && (b < a + c) && (c < a + b)) {
                        if ((a == b) && (b == c)) {
                            printf("Equilateral triangle");
                        } else if ((a != b) && (a != c) && (b != c)) {
                            printf("Scalene triangle");
                        } else {
                            printf("Isosceles triangle");
                        }
                    } else {
                        printf("Triangle cannot be formed");
                    }
            
                    printf("\nDo you want to check another triangle? (y/n): \n");
                    choice = getch();
            
                    if (tolower(choice) != 'y') {
                        break;
                    }
                }
            
                return 0;
            }
