        #include<stdio.h>
        #include<stdlib.h>
            
        int main() {
            int a[20], n, low, high, mid, key;
            char ch = 'y';
        
            while (ch == 'y' || ch == 'Y') {
                int flag = 0;
                printf("Enter the value of n: ");
                scanf("%d", &n);
                
                if(n==0){
                    printf("N is invalid and key not found");
                    exit(0);
                }

                if (n > 0) {
                    printf("Enter %d elements in ASCENDING order\n", n);
                    for (int i = 0; i < n; i++) {
                        scanf("%d", &a[i]);
                    }
        
                    printf("Enter the key element to be searched: ");
                    scanf("%d", &key);
        
                    low = 0;
                    high = n - 1;
        
                    while (low <= high) {
                        mid = (low + high) / 2;
                        if (a[mid] == key) {
                            flag = 1;
                            break;
                        } else if (a[mid] < key) {
                            low = mid + 1;
                        } else {
                            high = mid - 1;
                        }
                    }
        
                    if (flag == 1)
                        printf("Successful search\nElement found at Location %d\n", mid + 1);
                    else
                        printf("Key element is not found\n");
                } else {
                    printf("Wrong input\n");
                }
        
                printf("Do you want to continue (y/n): ");
                scanf(" %c", &ch);
                printf("\n\n");
            }
        
            return 0;
        }

        #include <stdio.h>
        #include <ctype.h>
        #include <conio.h>
        #include<stdlib.h>
            
        void clear_input_buffer() {
            int ch;
            while ((ch = getchar()) != '\n' && ch != EOF);
        }
        
        int main() {
            int a, b, c;
            char choice;
        
            while (1) {
                int validInput = 1;
                printf("Enter three sides of the triangle (a:, b:, c:): \n");
        
                printf("a: ");
                if (scanf("%d", &a) != 1) {
                    printf("Invalid input for side 'a'\n");
                    validInput = 0;
                    clear_input_buffer();
                }
        
                if (validInput) {
                    printf("b: ");
                    if (scanf("%d", &b) != 1) {
                        printf("Invalid input for side 'b'\n");
                        validInput = 0;
                        clear_input_buffer();
                    }
                }
        
                if (validInput) {
                    printf("c: ");
                    if (scanf("%d", &c) != 1) {
                        printf("Invalid input for side 'c'\n");
                        validInput = 0;
                        clear_input_buffer();
                    }
                }
        
                if (!validInput) {
                    printf("Not a triangle\n");
                    continue;
                }
        
                int valid = 1;
        
                if (a < 1 || a > 10) {
                    printf("Side 'a' is not in range (1-10)\n");
                    valid = 0;
                }
                if (b < 1 || b > 10) {
                    printf("Side 'b' is not in range (1-10)\n");
                    valid = 0;
                }
                if (c < 1 || c > 10) {
                    printf("Side 'c' is not in range (1-10)\n");
                    valid = 0;
                }
        
                if (valid) {
                    if ((a < b + c) && (b < a + c) && (c < a + b)) {
                        if ((a == b) && (b == c)) {
                            printf("Equilateral triangle");
                        } else if ((a != b) && (a != c) && (b != c)) {
                            printf("Scalene triangle");
                        } else {
                            printf("Isosceles triangle");
                        }
                    } else {
                        printf("Triangle cannot be formed");
                    }
                } else {
                    printf("Not a valid triangle\n");
                }
        
                printf("\nDo you want to check another triangle? (y/n): \n");
                choice = getch();
        
                if (tolower(choice) != 'y') {
                    break;
                }
            }
        
            return 0;
        }
